55static int dma_heap_open(void)
56{
57 static const char *const heaps[] = {
58 "/dev/dma_heap/system-uncached",
59 "/dev/dma_heap/system",
60 NULL
61 };
62
63 for (int i = 0; heaps[i]; i++) {
64 int fd = open(heaps[i], O_RDONLY | O_CLOEXEC);
65 if (fd >= 0)
66 return fd;
67 }
68 return -1;
69}

71static int dma_heap_alloc(int size)
72{
73 int heap_fd = dma_heap_open();
74 if (heap_fd < 0)
75 return -1;
76
77 struct dma_heap_allocation_data alloc = {0};
78 alloc.len = (uint64_t)size;
79 alloc.fd_flags = O_RDWR | O_CLOEXEC;
80
81 if (ioctl(heap_fd, DMA_HEAP_IOCTL_ALLOC, &alloc) < 0) {
82 close(heap_fd);
83 return -1;
84 }
85
86 close(heap_fd);
87 return (int)alloc.fd;
88}

90rkvc_err rkvc_avframe_alloc_contiguous(AVFrame *av_frame)
91{
92 if (!av_frame || av_frame->width <= 0 || av_frame->height <= 0)
93 return RKVC_ERR_INVALID;
94
95 enum AVPixelFormat av_fmt = (enum AVPixelFormat)av_frame->format;
96 if (av_fmt == AV_PIX_FMT_NONE)
97 return RKVC_ERR_INVALID;

100 av_frame->width,

236 b->strides[i] = (uint32_t)b->av_frame->linesize[i];

272rkvc_buffer *rkvc_buffer_from_drm_frame(AVFrame *hw_frame)
273{
274 if (!hw_frame || hw_frame->format != AV_PIX_FMT_DRM_PRIME)
275 return NULL;
276
277 AVDRMFrameDescriptor *desc = (AVDRMFrameDescriptor *)hw_frame->data[0];
278 if (!desc || desc->nb_objects == 0 || desc->nb_layers == 0)
279 return NULL;
280
281 const AVDRMLayerDescriptor *layer = &desc->layers[0];
282 /* 线性 NV12：双 plane；AFBC/RFBC 为单 plane，需 hw download。 */
283 if (layer->format != DRM_FORMAT_NV12 || layer->nb_planes < 2)
284 return NULL;
285
286 const int pitch = layer->planes[0].pitch;
287 if (pitch <= 0)
288 return NULL;
289
290 rkvc_buffer *b = rkvc_buffer_wrap_avframe(hw_frame, 1);
291 if (!b)
292 return NULL;
293
294 b->mem_type = RKVC_MEM_DMABUF;
295 b->fd = desc->objects[0].fd;
296 b->modifier = desc->objects[0].format_modifier;
297 b->format = RKVC_PIX_FMT_NV12;
298 b->width = (uint32_t)hw_frame->width;
299 b->height = (uint32_t)hw_frame->height;
300 b->strides[0] = (uint32_t)pitch;
301 b->strides[1] = (uint32_t)pitch;
302 if (b->av_frame) {
303 b->av_frame->linesize[0] = pitch;
304 b->av_frame->linesize[1] = pitch;
305 }
306 return b;
307}

351static void buffer_free(rkvc_buffer *b)
352{
353 if (b->kind == RKVC_BUF_VIDEO) {
354 int close_fd = -1;
355 if (b->owns_avframe && b->av_frame) {
356 /*
357 * RKMPP DRM 帧的 fd 由 AVFrame/MPP 池管理，av_frame_free 会归还；
358 * 仅对 dma-heap 自分配缓冲（NV12 等）在释放后 close fd。
359 */
360 if (b->mem_type == RKVC_MEM_DMABUF && b->fd >= 0 &&
361 b->av_frame->format != AV_PIX_FMT_DRM_PRIME)
362 close_fd = b->fd;
363 av_frame_free(&b->av_frame);
364 if (close_fd >= 0)
365 close(close_fd);
366 }
367 } else if (b->kind == RKVC_BUF_BITSTREAM) {
368 if (b->owns_data)
369 rkvc_free(b->data);
370 }
371 pthread_mutex_destroy(&b->lock);
372 rkvc_free(b);
373}
