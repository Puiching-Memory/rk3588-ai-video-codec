104static int rkvc_dev_access(const char *path, int mode)
105{

144{

194static int mpp_default_dma_heap_accessible(void)
195{
196 /*
197 * rockchip-mpp opens type 0 as /dev/dma_heap/system-uncached. If that
198 * fails, it only remaps type 0 by flipping cacheable/dma32 flags, so cma*
199 * heaps are not valid fallbacks for the default encoder/decoder path.
200 */
201 static const char *const names[] = {
202 "system-uncached",
203 "system",
204 "system-uncached-dma32",
205 "system-dma32",
206 NULL
207 };
208
209 for (int i = 0; names[i]; i++) {
210 char path[PATH_MAX];
211
212 int written = snprintf(path, sizeof(path), "/dev/dma_heap/%s",
213 names[i]);
214 if (written < 0 || (size_t)written >= sizeof(path))
215 continue;
216
217 if (path_openable(path, O_RDONLY))
218 return 1;
219 }
220
221 return 0;
222}

224rkvc_err rkvc_check_hw_permissions(void)
225{
226 if (!mpp_codec_device_accessible()) {
227 RKVC_LOG("MPP codec device permission denied: /dev/mpp_service or legacy codec nodes");
228 return RKVC_ERR_PERMISSION;
229 }
230
231 if (dma_heap_runtime_selected()) {
232 if (!mpp_default_dma_heap_accessible()) {
233 RKVC_LOG("MPP default DMA heap permission denied: /dev/dma_heap/system-uncached");
234 return RKVC_ERR_PERMISSION;
235 }
236 return RKVC_OK;
237 }
238
239 if (!drm_allocator_accessible() && !ion_allocator_accessible()) {
240 RKVC_LOG("DMA-BUF allocator permission denied: /dev/dma_heap/* or /dev/dri/*");
241 return RKVC_ERR_PERMISSION;
242 }
243
244 return RKVC_OK;
245}

247rkvc_err rkvc_query_caps(rkvc_caps *caps)
248{
249 if (!caps)
250 return RKVC_ERR_INVALID;
251
252 memset(caps, 0, sizeof(*caps));
253
254 rkvc_err hw_perm = rkvc_check_hw_permissions();
255 int hw_usable = (hw_perm == RKVC_OK);
256
257 caps->has_h264_enc = (avcodec_find_encoder_by_name("h264_rkmpp") != NULL
258 && hw_usable);
259 caps->has_hevc_enc = (avcodec_find_encoder_by_name("hevc_rkmpp") != NULL
260 && hw_usable);
261 caps->has_av1_enc = 1; /* SVT-AV1 submodule */
262
263 caps->has_h264_dec = (avcodec_find_decoder_by_name("h264_rkmpp") != NULL
264 && hw_usable);
265 caps->has_hevc_dec = (avcodec_find_decoder_by_name("hevc_rkmpp") != NULL
266 && hw_usable);
267 caps->has_av1_dec = (avcodec_find_decoder_by_name("av1_rkmpp") != NULL
268 && hw_usable);
269
270 caps->has_dma_heap = mpp_default_dma_heap_accessible();
271 caps->has_rga = (rkvc_dev_access("/dev/rga", R_OK | W_OK) == 0);
272
273 /* RK3588 最大分辨率 */
274 caps->max_width = 7680;
275 caps->max_height = 4320;
276
277 return RKVC_OK;
278}
