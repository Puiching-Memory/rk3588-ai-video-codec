23static rkvc_err svt_check(EbErrorType ret)
24{
25 if (ret == EB_ErrorNone)

28 return RKVC_ERR_AGAIN;

34static rkvc_err svt_alloc_io(rkvc_svt_enc *enc)
35{
36 const int w = enc->width;
37 const int h = enc->height;
38 const size_t y = (size_t)w * (size_t)h;
39 const size_t c = y / 4;
40
41 enc->in_header = rkvc_calloc(1, sizeof(*enc->in_header));
42 enc->in_io = rkvc_calloc(1, sizeof(*enc->in_io));
43 if (!enc->in_header || !enc->in_io)
44 return RKVC_ERR_NOMEM;
45
46 enc->in_io->luma = rkvc_malloc(y);
47 enc->in_io->cb = rkvc_malloc(c);
48 enc->in_io->cr = rkvc_malloc(c);
49 if (!enc->in_io->luma || !enc->in_io->cb || !enc->in_io->cr)
50 return RKVC_ERR_NOMEM;
51
52 enc->in_io->y_stride = (uint32_t)w;
53 enc->in_io->cb_stride = (uint32_t)(w / 2);
54 enc->in_io->cr_stride = (uint32_t)(w / 2);
55 enc->in_header->size = sizeof(*enc->in_header);
56 enc->in_header->p_buffer = (uint8_t *)enc->in_io;
57 return RKVC_OK;
58}

60static void copy_to_svt(rkvc_svt_enc *enc, const AVFrame *frame)
61{
62 const int w = enc->width;
63 const int h = enc->height;
64 EbSvtIOFormat *io = enc->in_io;
65
66 for (int y = 0; y < h; y++)
67 memcpy(io->luma + y * io->y_stride,
68 frame->data[0] + y * frame->linesize[0], (size_t)w);
69
70 for (int y = 0; y < h / 2; y++) {
71 if (frame->format == AV_PIX_FMT_YUV420P) {
72 memcpy(io->cb + y * io->cb_stride,
73 frame->data[1] + y * frame->linesize[1], (size_t)(w / 2));
74 memcpy(io->cr + y * io->cr_stride,
75 frame->data[2] + y * frame->linesize[2], (size_t)(w / 2));
76 } else {
77 const uint8_t *uv = frame->data[1] + y * frame->linesize[1];
78 for (int x = 0; x < w / 2; x++) {
79 io->cb[y * io->cb_stride + x] = uv[x * 2];
80 io->cr[y * io->cr_stride + x] = uv[x * 2 + 1];
81 }
82 }
83 }
84}

86rkvc_err rkvc_svt_enc_open(rkvc_svt_enc **out, const rkvc_svt_enc_config *cfg)
87{
88 if (!out || !cfg || cfg->width <= 0 || cfg->height <= 0)
89 return RKVC_ERR_INVALID;
90
91 *out = NULL;
92 rkvc_init();
93
94 rkvc_svt_enc *enc = rkvc_calloc(1, sizeof(*enc));
95 if (!enc)
96 return RKVC_ERR_NOMEM;
97
98 enc->width = cfg->width;
99 enc->height = cfg->height;
100 enc->input_format = cfg->input_format;
101
102 EbErrorType ret = svt_av1_enc_init_handle(&enc->handle, &enc->cfg);
103 if (ret != EB_ErrorNone) {
104 rkvc_svt_enc_close(enc);
105 return RKVC_ERR_NOT_FOUND;
106 }
107
108 enc->cfg.source_width = (uint32_t)cfg->width;
109 enc->cfg.source_height = (uint32_t)cfg->height;
110 enc->cfg.frame_rate_numerator = (uint32_t)cfg->fps_num;
111 enc->cfg.frame_rate_denominator = (uint32_t)cfg->fps_den;
112 enc->cfg.encoder_color_format = EB_YUV420;
113 enc->cfg.encoder_bit_depth = 8;
114 enc->cfg.enc_mode = (uint8_t)(cfg->svt_preset > 0 ? cfg->svt_preset
115 : RKVC_SVT_PRESET_PERF);
116 enc->cfg.level_of_parallelism = RKVC_SVT_DEFAULT_LP;
117 enc->cfg.target_bit_rate = (uint32_t)cfg->bitrate;
118 switch (cfg->rc_mode) {
119 case RKVC_RC_CBR:
120 /* SVT 文件/转码为 random-access，CBR 会初始化失败；对齐为 VBR。 */
121 enc->cfg.rate_control_mode = SVT_AV1_RC_MODE_VBR;
122 break;
123 case RKVC_RC_CQP:
124 enc->cfg.rate_control_mode = SVT_AV1_RC_MODE_CQP_OR_CRF;
125 break;
126 case RKVC_RC_VBR:
127 default:
128 enc->cfg.rate_control_mode = SVT_AV1_RC_MODE_VBR;
129 break;
130 }
131 if (cfg->gop_size > 1)
132 enc->cfg.intra_period_length = cfg->gop_size - 1;
133
134 ret = svt_av1_enc_set_parameter(enc->handle, &enc->cfg);
135 if (ret != EB_ErrorNone) {
136 rkvc_svt_enc_close(enc);
137 return RKVC_ERR_INVALID;
138 }
139
140 ret = svt_av1_enc_init(enc->handle);
141 if (ret != EB_ErrorNone) {
142 rkvc_svt_enc_close(enc);
143 return RKVC_ERR_INTERNAL;
144 }
145
146 if (svt_alloc_io(enc) != RKVC_OK) {
147 rkvc_svt_enc_close(enc);
148 return RKVC_ERR_NOMEM;
149 }
150
151 *out = enc;
152 return RKVC_OK;
153}

174rkvc_err rkvc_svt_enc_write_header(rkvc_svt_enc *enc, AVCodecParameters *par)
175{
176 if (!enc || !enc->handle || !par)
177 return RKVC_ERR_INVALID;
178
179 EbBufferHeaderType *hdr = NULL;
180 EbErrorType ret = svt_av1_enc_stream_header(enc->handle, &hdr);
181 if (ret != EB_ErrorNone || !hdr || hdr->n_filled_len == 0)
182 return RKVC_ERR_MUX;
183
184 par->extradata = av_mallocz(hdr->n_filled_len + AV_INPUT_BUFFER_PADDING_SIZE);
185 if (!par->extradata) {
186 svt_av1_enc_stream_header_release(hdr);
187 return RKVC_ERR_NOMEM;
188 }
189 memcpy(par->extradata, hdr->p_buffer, hdr->n_filled_len);
190 par->extradata_size = (int)hdr->n_filled_len;
191 svt_av1_enc_stream_header_release(hdr);
192 return RKVC_OK;
193}

195rkvc_err rkvc_svt_enc_send_frame(rkvc_svt_enc *enc, rkvc_buffer *frame)
196{
197 if (!enc || !enc->handle)
198 return RKVC_ERR_INVALID;
199 if (enc->flushed)
200 return RKVC_ERR_EOF;
201 if (!frame)
202 return rkvc_svt_enc_drain(enc);
203
204 rkvc_buffer *host = NULL;
205 const rkvc_buffer *work = frame;
206 if (frame->mem_type == RKVC_MEM_DMABUF) {
207 rkvc_err err = rkvc_dma_to_host(frame, &host);
208 if (err != RKVC_OK)
209 return err;
210 work = host;
211 }
212
213 if (!work->av_frame) {
214 rkvc_buffer_unref(host);
215 return RKVC_ERR_INVALID;
216 }
217
218 copy_to_svt(enc, work->av_frame);
219 rkvc_buffer_unref(host);
220
221 const uint32_t y_size = (uint32_t)((size_t)enc->width * (size_t)enc->height);
222 enc->in_header->n_filled_len = y_size + 2 * (y_size / 4);
223 enc->in_header->pts = work->pts >= 0 ? work->pts : enc->next_pts++;
224 enc->in_header->dts = enc->in_header->pts;
225
226 EbErrorType ret = svt_av1_enc_send_picture(enc->handle, enc->in_header);
227 return svt_check(ret);
228}

230rkvc_err rkvc_svt_enc_receive_packet(rkvc_svt_enc *enc, rkvc_buffer **out)
231{
232 if (!enc || !enc->handle || !out)
233 return RKVC_ERR_INVALID;
234
235 *out = NULL;
236 EbBufferHeaderType *pkt = NULL;
237 EbErrorType ret = svt_av1_enc_get_packet(enc->handle, &pkt,
238 (uint8_t)enc->pic_done);
239 rkvc_err err = svt_check(ret);
240 if (err != RKVC_OK)
241 return err;
242 if (!pkt)
243 return enc->pic_done ? RKVC_ERR_EOF : RKVC_ERR_AGAIN;
244
245 if (pkt->flags & EB_BUFFERFLAG_EOS) {
246 svt_av1_enc_release_out_buffer(&pkt);
247 return RKVC_ERR_EOF;
248 }
249 if (pkt->n_filled_len == 0) {
250 svt_av1_enc_release_out_buffer(&pkt);
251 return RKVC_ERR_AGAIN;
252 }
253
254 rkvc_buffer *b = rkvc_calloc(1, sizeof(*b));
255 if (!b) {
256 svt_av1_enc_release_out_buffer(&pkt);
257 return RKVC_ERR_NOMEM;
258 }
259 b->kind = RKVC_BUF_BITSTREAM;
260 pthread_mutex_init(&b->lock, NULL);
261 b->ref_count = 1;
262 b->data = rkvc_malloc(pkt->n_filled_len);
263 if (!b->data) {
264 svt_av1_enc_release_out_buffer(&pkt);
265 rkvc_buffer_unref(b);
266 return RKVC_ERR_NOMEM;
267 }
268 memcpy(b->data, pkt->p_buffer, pkt->n_filled_len);
269 b->size = pkt->n_filled_len;
270 b->owns_data = 1;
271 b->pts = pkt->pts;
272 b->dts = pkt->dts;
273 b->key_frame = (pkt->pic_type == EB_AV1_KEY_PICTURE ||
274 pkt->pic_type == EB_AV1_INTRA_ONLY_PICTURE) ? 1 : 0;
275
276 svt_av1_enc_release_out_buffer(&pkt);
277 *out = b;
278 return RKVC_OK;
279}
