42static rkvc_err decoder_open_internal(rkvc_decoder **out,
43 const rkvc_decoder_config *cfg,
44 const char *input_path)
45{
46 if (!out)
47 return RKVC_ERR_INVALID;
48
49 *out = NULL;
50
51 rkvc_err err = validate_decoder_config(cfg);
52 if (err != RKVC_OK)
53 return err;
54
55 rkvc_init();
56
57 rkvc_decoder *dec = rkvc_calloc(1, sizeof(*dec));
58 if (!dec)
59 return RKVC_ERR_NOMEM;
60
61 dec->config = *cfg;
62 dec->video_stream_idx = -1;
63 pthread_mutex_init(&dec->lock, NULL);
64
65 /*
66 * 不预设 hw_device_ctx。
67 * RKMPP 解码器在 avcodec_open2 时自动创建硬件设备上下文。
68 */
69
70 /* 查找解码器 */
71 const AVCodec *codec = avcodec_find_decoder_by_name("hevc_rkmpp");
72 if (!codec) {
73 codec = avcodec_find_decoder(AV_CODEC_ID_HEVC);
74 if (!codec) {
75 rkvc_decoder_close(dec);
76 return RKVC_ERR_NOT_FOUND;
77 }
78 RKVC_LOG("hevc_rkmpp not found, using software: %s", codec->name);
79 }
80
81 /* 文件模式 */
82 if (input_path) {
83 int ret = avformat_open_input(&dec->fmt_ctx, input_path, NULL, NULL);
84 if (ret < 0) {
85 rkvc_decoder_close(dec);
86 return rkvc_from_averror(ret);
87 }
88
89 ret = avformat_find_stream_info(dec->fmt_ctx, NULL);
90 if (ret < 0) {
91 rkvc_decoder_close(dec);
92 return rkvc_from_averror(ret);
93 }
94
95 /* 找视频流 */
96 for (unsigned i = 0; i < dec->fmt_ctx->nb_streams; i++) {
97 if (dec->fmt_ctx->streams[i]->codecpar->codec_type ==
98 AVMEDIA_TYPE_VIDEO) {
99 dec->video_stream_idx = (int)i;
100 break;
101 }
102 }
103 if (dec->video_stream_idx < 0) {
104 rkvc_decoder_close(dec);
105 return RKVC_ERR_NOT_FOUND;
106 }
107
108 /* 创建解码器上下文 */
109 dec->codec_ctx = avcodec_alloc_context3(codec);
110 if (!dec->codec_ctx) {
111 rkvc_decoder_close(dec);
112 return RKVC_ERR_NOMEM;
113 }
114
115 ret = avcodec_parameters_to_context(
116 dec->codec_ctx,
117 dec->fmt_ctx->streams[dec->video_stream_idx]->codecpar);
118 if (ret < 0) {
119 rkvc_decoder_close(dec);
120 return rkvc_from_averror(ret);
121 }
122
123 dec->file_mode = 1;
124 } else {
125 /* 无文件模式 */
126 dec->codec_ctx = avcodec_alloc_context3(codec);
127 if (!dec->codec_ctx) {
128 rkvc_decoder_close(dec);
129 return RKVC_ERR_NOMEM;
130 }
131 }
132
133 /*
134 * 不手动设置 hw_device_ctx。
135 * RKMPP 解码器会自动创建/复用硬件设备上下文。
136 * 手动设置反而会导致 RKMPP 初始化时函数指针为 NULL 的崩溃。
137 */
138
139 /* 解码器参数 */
140 if (cfg->low_delay)
141 dec->codec_ctx->flags |= AV_CODEC_FLAG_LOW_DELAY;
142
143 if (cfg->threads > 0)
144 dec->codec_ctx->thread_count = cfg->threads;
145 else
146 dec->codec_ctx->thread_count = 1; /* RKMPP 单线程 */
147
148 /* 打开解码器 */
149 int ret = avcodec_open2(dec->codec_ctx, codec, NULL);
150 if (ret < 0) {
151 RKVC_LOG("avcodec_open2 decoder failed: %s", av_err2str(ret));
152 rkvc_decoder_close(dec);
153 return rkvc_from_averror(ret);
154 }
155
156 /* 分配 packet */
157 dec->pkt = av_packet_alloc();
158 if (!dec->pkt) {
159 rkvc_decoder_close(dec);
160 return RKVC_ERR_NOMEM;
161 }
162
163 *out = dec;
164 return RKVC_OK;
165}

184rkvc_err rkvc_decoder_send_packet(rkvc_decoder *dec,
185 const uint8_t *data, int size,
186 int64_t pts, int64_t dts)
187{
188 if (!dec || !dec->codec_ctx)
189 return RKVC_ERR_INVALID;
190
191 if (dec->flushed)
192 return RKVC_ERR_EOF;
193
194 if (!data || size <= 0) {
195 /* flush */
196 return rkvc_decoder_drain(dec);
197 }
198
199 av_packet_unref(dec->pkt);
200 dec->pkt->data = (uint8_t *)data;
201 dec->pkt->size = size;
202 dec->pkt->pts = pts;
203 dec->pkt->dts = dts;
204
205 int ret = avcodec_send_packet(dec->codec_ctx, dec->pkt);
206 /* 不 unref，data 是外部持有的 */
207
208 if (ret < 0) {
209 if (ret == AVERROR(EAGAIN))
210 return RKVC_ERR_AGAIN;
211 return rkvc_from_averror(ret);
212 }
213
214 return RKVC_OK;
215}

219rkvc_err rkvc_decoder_read_packet(rkvc_decoder *dec)
220{
221 if (!dec || !dec->fmt_ctx)
222 return RKVC_ERR_INVALID;
223
224 if (dec->flushed)
225 return RKVC_ERR_EOF;
226
227 av_packet_unref(dec->pkt);
228
229 int ret;
230 for (;;) {
231 ret = av_read_frame(dec->fmt_ctx, dec->pkt);
232 if (ret < 0) {
233 if (ret == AVERROR_EOF) {
234 dec->flushed = 1;
235 /* 发送 flush 包 */
236 avcodec_send_packet(dec->codec_ctx, NULL);
237 return RKVC_ERR_EOF;
238 }
239 return rkvc_from_averror(ret);
240 }
241
242 /* 跳过非视频流 */
243 if (dec->pkt->stream_index == dec->video_stream_idx)
244 break;
245
246 av_packet_unref(dec->pkt);
247 }
248
249 ret = avcodec_send_packet(dec->codec_ctx, dec->pkt);
250 av_packet_unref(dec->pkt);
251
252 if (ret < 0) {
253 if (ret == AVERROR(EAGAIN))
254 return RKVC_ERR_AGAIN;
255 return rkvc_from_averror(ret);
256 }
257
258 return RKVC_OK;
259}

263rkvc_err rkvc_decoder_receive_frame(rkvc_decoder *dec, rkvc_frame **f)
264{
265 if (!dec || !dec->codec_ctx || !f)
266 return RKVC_ERR_INVALID;
267
268 *f = NULL;
269
270 AVFrame *frame = av_frame_alloc();
271 if (!frame)
272 return RKVC_ERR_NOMEM;
273
274 int ret = avcodec_receive_frame(dec->codec_ctx, frame);
275 if (ret < 0) {
276 av_frame_free(&frame);
277 if (ret == AVERROR(EAGAIN))
278 return RKVC_ERR_AGAIN;
279 if (ret == AVERROR_EOF)
280 return RKVC_ERR_EOF;
281 return rkvc_from_averror(ret);
282 }
283
284 /* 如果是硬件帧，下载到软件 */
285 if (frame->format == AV_PIX_FMT_RKMPP) {
286 AVFrame *sw_frame = av_frame_alloc();
287 if (!sw_frame) {
288 av_frame_free(&frame);
289 return RKVC_ERR_NOMEM;
290 }
291
292 ret = av_hwframe_transfer_data(sw_frame, frame, 0);
293 av_frame_free(&frame);
294
295 if (ret < 0) {
296 av_frame_free(&sw_frame);
297 return rkvc_from_averror(ret);
298 }
299
300 frame = sw_frame;
301 }
302
303 /* 包装成 rkvc_frame */
304 rkvc_frame *rkf = rkvc_frame_wrap_avframe(frame);
305 if (!rkf) {
306 av_frame_free(&frame);
307 return RKVC_ERR_NOMEM;
308 }
309
310 *f = rkf;
311 return RKVC_OK;
312}

316rkvc_err rkvc_decoder_drain(rkvc_decoder *dec)
317{
318 if (!dec || !dec->codec_ctx)
319 return RKVC_ERR_INVALID;
320
321 if (dec->flushed)
322 return RKVC_OK;
323
324 int ret = avcodec_send_packet(dec->codec_ctx, NULL);
325 if (ret < 0 && ret != AVERROR_EOF)
326 return rkvc_from_averror(ret);
327
328 dec->flushed = 1;
329 return RKVC_OK;
330}

335{

340 av_packet_free(&dec->pkt);
341 if (dec->codec_ctx)
342 avcodec_free_context(&dec->codec_ctx);
343 if (dec->fmt_ctx)

355rkvc_err rkvc_decoder_get_video_info(const rkvc_decoder *dec,
356 int *width, int *height,
357 int *fps_num, int *fps_den)
358{
359 if (!dec)
360 return RKVC_ERR_INVALID;
361
362 if (dec->codec_ctx) {
363 if (width) *width = dec->codec_ctx->width;
364 if (height) *height = dec->codec_ctx->height;
365 }
366
367 if (dec->fmt_ctx && dec->video_stream_idx >= 0) {
368 AVStream *st = dec->fmt_ctx->streams[dec->video_stream_idx];
369 if (fps_num && fps_den) {
370 AVRational r = av_guess_frame_rate(dec->fmt_ctx, st, NULL);
371 *fps_num = r.num;
372 *fps_den = r.den;
373 }
374 }
375
376 return RKVC_OK;
377}
