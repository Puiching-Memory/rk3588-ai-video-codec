119rkvc_err rkvc_post_upscale_buffer(const rkvc_buffer *src, rkvc_buffer **dst,
120 int dst_w, int dst_h,
121 rkvc_upscale_algo algo)
122{
123 if (!src || !dst || dst_w <= 0 || dst_h <= 0)
124 return RKVC_ERR_INVALID;
125 if (algo == RKVC_UPSCALE_NONE)
126 return RKVC_ERR_INVALID;
127 if (!rkvc_rga_available())
128 return RKVC_ERR_HW;
129
130 *dst = NULL;
131
132 if (!src->av_frame)
133 return RKVC_ERR_INVALID;
134
135 const int sw = src->av_frame->width;
136 const int sh = src->av_frame->height;
137 const rkvc_pix_fmt src_fmt = src->format;
138 const rkvc_pix_fmt dst_fmt = src_fmt;
139
140 if (sw == dst_w && sh == dst_h) {
141 *dst = rkvc_buffer_ref((rkvc_buffer *)src);
142 return RKVC_OK;
143 }
144
145 rkvc_buffer *nv12_src = NULL;
146 rkvc_buffer *scaled = NULL;
147 rkvc_err err = RKVC_OK;
148
149 if (src_fmt == RKVC_PIX_FMT_NV12) {
150 nv12_src = rkvc_buffer_ref((rkvc_buffer *)src);
151 } else if (src_fmt == RKVC_PIX_FMT_YUV420P) {
152 err = yuv420p_to_nv12_buffer(src, &nv12_src);
153 } else {
154 return RKVC_ERR_FORMAT;
155 }
156
157 if (err == RKVC_OK)
158 err = rkvc_rga_scale_buffer(nv12_src, &scaled, dst_w, dst_h,
159 RKVC_PIX_FMT_NV12, algo);
160
161 rkvc_buffer_unref(nv12_src);
162
163 if (err != RKVC_OK)
164 return err;
165
166 if (dst_fmt == RKVC_PIX_FMT_YUV420P) {
167 err = nv12_to_yuv420p_buffer(scaled, dst);
168 rkvc_buffer_unref(scaled);
169 return err;
170 }
171
172 scaled->pts = src->pts;
173 *dst = scaled;
174 return RKVC_OK;
175}
