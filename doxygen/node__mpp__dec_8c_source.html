181rkvc_err rkvc_mpp_dec_receive_frame(rkvc_mpp_dec *dec, rkvc_buffer **out)
182{
183 if (!dec || !dec->ctx || !out)
184 return RKVC_ERR_INVALID;
185
186 *out = NULL;
187 AVFrame *frame = av_frame_alloc();
188 if (!frame)
189 return RKVC_ERR_NOMEM;
190
191 int ret = avcodec_receive_frame(dec->ctx, frame);
192 if (ret < 0) {
193 av_frame_free(&frame);
194 if (ret == AVERROR(EAGAIN))
195 return RKVC_ERR_AGAIN;
196 if (ret == AVERROR_EOF)
197 return RKVC_ERR_EOF;
198 return rkvc_from_averror(ret);
199 }
200
201 if (frame->format == AV_PIX_FMT_DRM_PRIME) {
202 rkvc_buffer *dma = NULL;
203 if (dec->output_format == RKVC_PIX_FMT_NV12)
204 dma = rkvc_buffer_from_drm_frame(frame);
205 if (dma) {
206 dma->pts = frame->pts;
207 *out = dma;
208 return RKVC_OK;
209 }
210 rkvc_buffer_unref(dma);
211
212 AVFrame *sw = av_frame_alloc();
213 if (!sw) {
214 av_frame_free(&frame);
215 return RKVC_ERR_NOMEM;
216 }
217 sw->format = AV_PIX_FMT_NV12;
218 ret = av_hwframe_transfer_data(sw, frame, 0);
219 av_frame_free(&frame);
220 if (ret < 0) {
221 av_frame_free(&sw);
222 return rkvc_from_averror(ret);
223 }
224 frame = sw;
225 }
226
227 enum AVPixelFormat want = rkvc_to_av_pix_fmt(dec->output_format);
228 if (frame->format != want)
229 return dec_sws_to_format(dec, frame, out);
230
231 *out = rkvc_buffer_wrap_avframe(frame, 1);
232 if (!*out) {
233 av_frame_free(&frame);
234 return RKVC_ERR_NOMEM;
235 }
