36static rkvc_err validate_stream_config(const rkvc_stream_config *cfg)
37{
38 if (!cfg)
39 return RKVC_ERR_INVALID;
40
41 if (!rkvc_is_valid_stream_dir(cfg->direction))
42 return RKVC_ERR_INVALID;
43 if (cfg->buffer_size <= 0 || cfg->buffer_size > RKVC_STREAM_BUF_MAX)
44 return RKVC_ERR_INVALID;
45 if (cfg->drop_frames != 0 && cfg->drop_frames != 1)
46 return RKVC_ERR_INVALID;
47
48 if (cfg->direction == RKVC_STREAM_ENCODE) {
49 if (cfg->width <= 0 || cfg->height <= 0)
50 return RKVC_ERR_INVALID;
51 if (cfg->fps_num <= 0 || cfg->fps_den <= 0)
52 return RKVC_ERR_INVALID;
53 if (cfg->bitrate <= 0)
54 return RKVC_ERR_INVALID;
55 if (!rkvc_is_valid_pix_fmt(cfg->input_format) ||
56 !rkvc_is_valid_preset(cfg->preset))
57 return RKVC_ERR_INVALID;
58 } else if (!rkvc_is_valid_pix_fmt(cfg->output_format)) {
59 return RKVC_ERR_INVALID;
60 }
61
62 return RKVC_OK;
63}

65rkvc_err rkvc_stream_open(rkvc_stream **out,
66 const rkvc_stream_config *cfg)
67{
68 if (!out)
69 return RKVC_ERR_INVALID;
70
71 *out = NULL;
72
73 rkvc_err err = validate_stream_config(cfg);
74 if (err != RKVC_OK)
75 return err;
76
77 rkvc_stream *s = rkvc_calloc(1, sizeof(*s));
78 if (!s)
79 return RKVC_ERR_NOMEM;
80
81 s->config = *cfg;
82 pthread_mutex_init(&s->buf_lock, NULL);
83 pthread_cond_init(&s->buf_not_full, NULL);
84 pthread_cond_init(&s->buf_not_empty, NULL);
85
86 if (cfg->direction == RKVC_STREAM_ENCODE) {
87 /* 创建内部编码器 */
88 rkvc_encoder_config enc_cfg = rkvc_encoder_config_defaults();
89 enc_cfg.width = cfg->width;
90 enc_cfg.height = cfg->height;
91 enc_cfg.fps_num = cfg->fps_num;
92 enc_cfg.fps_den = cfg->fps_den;
93 enc_cfg.bitrate = cfg->bitrate;
94 enc_cfg.input_format = cfg->input_format;
95 enc_cfg.preset = cfg->preset;
96
97 err = rkvc_encoder_open(&s->enc, &enc_cfg);
98 if (err != RKVC_OK) {
99 pthread_mutex_destroy(&s->buf_lock);
100 pthread_cond_destroy(&s->buf_not_full);
101 pthread_cond_destroy(&s->buf_not_empty);
102 rkvc_free(s);
103 return err;
104 }
105 } else {
106 /* 创建内部解码器 */
107 rkvc_decoder_config dec_cfg = rkvc_decoder_config_defaults();
108 dec_cfg.output_format = cfg->output_format;
109
110 err = rkvc_decoder_open(&s->dec, &dec_cfg);
111 if (err != RKVC_OK) {
112 pthread_mutex_destroy(&s->buf_lock);
113 pthread_cond_destroy(&s->buf_not_full);
114 pthread_cond_destroy(&s->buf_not_empty);
115 rkvc_free(s);
116 return err;
117 }
118 }
119
120 *out = s;
121 return RKVC_OK;
122}

126rkvc_err rkvc_stream_push(rkvc_stream *s, const void *data)
127{
128 if (!s || !data)
129 return RKVC_ERR_INVALID;
130
131 if (s->finished)
132 return RKVC_ERR_EOF;
133
134 if (s->config.direction == RKVC_STREAM_ENCODE) {
135 /* data 是 rkvc_frame* */
136 rkvc_frame *f = (rkvc_frame *)data;
137 rkvc_frame *to_encode = f;
138
139 /* 尺寸或格式不匹配时自动 RGA 缩放 */
140 int need_scale = (f->info.width != s->config.width) ||
141 (f->info.height != s->config.height) ||
142 (f->info.format != s->config.input_format);
143 if (need_scale) {
144 rkvc_scale_config sc = {
145 .dst_width = s->config.width,
146 .dst_height = s->config.height,
147 .dst_format = s->config.input_format,
148 };
149 rkvc_err sc_err = rkvc_frame_scale(f, &to_encode, &sc);
150 if (sc_err != RKVC_OK)
151 return sc_err;
152 }
153
154 rkvc_err err = rkvc_encoder_send_frame(s->enc, to_encode);
155
156 /* 编码器满时短暂等待后重试，让 MPP async 线程产出编码包 */
157 int retries = 0;
158 while (err == RKVC_ERR_AGAIN && retries < 1000) {
159 struct timespec ts = {0, 100000}; /* 100us */
160 nanosleep(&ts, NULL);
161 err = rkvc_encoder_send_frame(s->enc, to_encode);
162 retries++;
163 }
164
165 if (need_scale)
166 rkvc_frame_unref(to_encode);
167
168 if (err == RKVC_OK)
169 s->stats.frames_in++;
170
171 return err;
172 } else {
173 /* data 是 rkvc_packet* (压缩数据) */
174 const rkvc_packet *pkt = (const rkvc_packet *)data;
175
176 rkvc_err err = rkvc_decoder_send_packet(s->dec,
177 pkt->data, pkt->size,
178 pkt->pts, pkt->dts);
179 if (err == RKVC_OK)
180 s->stats.frames_in++;
181
182 return err;
183 }
184}

188rkvc_err rkvc_stream_pull(rkvc_stream *s, void *out, int timeout_ms)
189{
190 if (!s || !out)
191 return RKVC_ERR_INVALID;
192
193 (void)timeout_ms; /* 当前实现为非阻塞 */
194
195 if (s->config.direction == RKVC_STREAM_ENCODE) {
196 /* 输出 rkvc_packet (直接从编码器取) */
197 rkvc_packet *pkt = (rkvc_packet *)out;
198 rkvc_err err = rkvc_encoder_receive_packet(s->enc, pkt);
199 if (err == RKVC_OK) {
200 s->stats.frames_out++;
201 }
202 return err;
203 } else {
204 /* 输出 rkvc_frame* */
205 rkvc_frame **f = (rkvc_frame **)out;
206 rkvc_err err = rkvc_decoder_receive_frame(s->dec, f);
207 if (err == RKVC_OK) {
208 s->stats.frames_out++;
209 if (s->first_out_time == 0)
210 s->first_out_time = av_gettime_relative();
211 }
212 return err;
213 }
214}

255rkvc_err rkvc_stream_get_stats(const rkvc_stream *s,
256 rkvc_stream_stats *stats)
257{
258 if (!s || !stats)
259 return RKVC_ERR_INVALID;
260
261 *stats = s->stats;
262
263 /* 计算平均帧率 */
264 if (s->first_out_time > 0 && s->stats.frames_out > 1) {
265 int64_t now = av_gettime_relative();
266 double elapsed_s = (double)(now - s->first_out_time) / 1000000.0;
267 if (elapsed_s > 0)
268 stats->avg_fps = (double)(s->stats.frames_out - 1) / elapsed_s;
269 }
270
271 return RKVC_OK;
272}
