61rkvc_err rkvc_route_resolve(const rkvc_pipeline_desc *desc,
62 rkvc_route_plan *plan)
63{
64 if (!desc || !plan)
65 return RKVC_ERR_INVALID;
66
67 memset(plan, 0, sizeof(*plan));
68
69 if (desc->codec != RKVC_CODEC_AUTO) {
70 switch (desc->codec) {
71 case RKVC_CODEC_H264:
72 fill_mpp_h264(plan, "forced h264");
73 return RKVC_OK;
74 case RKVC_CODEC_HEVC:
75 fill_mpp_hevc(plan, "forced hevc");
76 return RKVC_OK;
77 case RKVC_CODEC_AV1:
78 fill_av1_svt(plan, RKVC_SVT_PRESET_PERF, "forced av1");
79 return RKVC_OK;
80 default:
81 return RKVC_ERR_INVALID;
82 }
83 }
84
85 const int pixels = desc->width * desc->height;
86 const int high_res = (pixels >= 1920 * 1080);
87
88 switch (desc->policy) {
89 case RKVC_POLICY_REALTIME:
90 fill_mpp_h264(plan, "realtime: h264_rkmpp E2E ~36fps@1080p");
91 break;
92
93 case RKVC_POLICY_BALANCED:
94 if (high_res && desc->fps_num >= 50)
95 fill_mpp_h264(plan, "balanced high-fps 1080p+: h264");
96 else
97 fill_mpp_hevc(plan, "balanced: hevc_rkmpp E2E ~27fps@1080p");
98 break;
99
100 case RKVC_POLICY_QUALITY:
101 fill_av1_svt(plan, RKVC_SVT_PRESET_PERF,
102 "quality: SVT preset 11 + av1_rkmpp");
103 break;
104
105 default:
106 return RKVC_ERR_INVALID;
107 }
108
109 return RKVC_OK;
