84/* ── FFmpeg 错误码映射 ─────────────────────────────────────────────── */
85
86rkvc_err rkvc_from_averror(int av_err)
87{
88 if (av_err >= 0)
89 return RKVC_OK;
90
91 switch (av_err) {
92 case AVERROR(ENOMEM): return RKVC_ERR_NOMEM;
93 case AVERROR(EINVAL): return RKVC_ERR_INVALID;
94 case AVERROR(ENOENT): return RKVC_ERR_NOT_FOUND;
95 case AVERROR(EIO): return RKVC_ERR_IO;
96 case AVERROR_EOF: return RKVC_ERR_EOF;
97 case AVERROR(EAGAIN): return RKVC_ERR_AGAIN;
98 case AVERROR(ENODEV): return RKVC_ERR_HW;
99 case AVERROR_UNKNOWN: return RKVC_ERR_INTERNAL;
100 default:
101 /* AVERROR_DECODER_NOT_FOUND / ENOMEM 等 */
102 if (av_err == AVERROR_DECODER_NOT_FOUND ||
103 av_err == AVERROR_ENCODER_NOT_FOUND)
104 return RKVC_ERR_NOT_FOUND;
105 return RKVC_ERR_INTERNAL;
106 }
107}

109/* ── RKMPP 硬件设备上下文单例 ──────────────────────────────────────── */
110
111static AVBufferRef *s_hw_device_ctx = NULL;
112static pthread_mutex_t s_hw_lock = PTHREAD_MUTEX_INITIALIZER;
113static int s_hw_init_ok = 0;
114
115rkvc_err rkvc_get_hw_device_ctx(AVBufferRef **out)
116{
117 if (!out)
118 return RKVC_ERR_INVALID;
119
120 pthread_mutex_lock(&s_hw_lock);
121
122 if (!s_hw_init_ok) {
123 int ret = av_hwdevice_ctx_create(&s_hw_device_ctx,
124 AV_HWDEVICE_TYPE_RKMPP,
125 NULL, NULL, 0);
126 if (ret < 0) {
127 RKVC_LOG("RKMPP hw device create failed: %s", av_err2str(ret));
128 s_hw_device_ctx = NULL;
129 pthread_mutex_unlock(&s_hw_lock);
130 return rkvc_from_averror(ret);
131 }
132 s_hw_init_ok = 1;
133 RKVC_LOG("RKMPP hw device created");
134 }
135
136 *out = av_buffer_ref(s_hw_device_ctx);
137 pthread_mutex_unlock(&s_hw_lock);
138
139 if (!*out)
140 return RKVC_ERR_NOMEM;
141 return RKVC_OK;
142}

218rkvc_frame *rkvc_frame_wrap_avframe(AVFrame *av_frame)
219{
220 if (!av_frame)
221 return NULL;
222
223 rkvc_frame *f = rkvc_calloc(1, sizeof(*f));
224 if (!f)
225 return NULL;
226
227 f->av_frame = av_frame;
228 f->ref_count = 1;
229 pthread_mutex_init(&f->lock, NULL);
230
231 /* 填充元数据 */
232 f->info.width = av_frame->width;
233 f->info.height = av_frame->height;
234 f->info.format = rkvc_from_av_pix_fmt(av_frame->format);
235 f->info.pts = av_frame->pts;
236 f->info.key_frame = (av_frame->flags & AV_FRAME_FLAG_KEY) != 0;
237
238 return f;
