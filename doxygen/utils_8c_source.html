87static int annexb_start_code_at(const uint8_t *data, size_t size,
88 size_t pos, size_t *nal_offset)
89{
90 if (pos + 4 <= size && data[pos] == 0x00 && data[pos + 1] == 0x00 &&
91 data[pos + 2] == 0x01) {
92 *nal_offset = pos + 3;
93 return 1;
94 }
95
96 if (pos + 5 <= size && data[pos] == 0x00 && data[pos + 1] == 0x00 &&
97 data[pos + 2] == 0x00 && data[pos + 3] == 0x01) {
98 *nal_offset = pos + 4;
99 return 1;
100 }
101
102 return 0;
103}

111static int is_h264_or_h265_annexb(const uint8_t *data, size_t size)
112{
113 size_t nal_offset = 0;
114 int h264_sps = 0;
115 int h264_pps = 0;
116 int h265_vps = 0;
117 int h265_sps = 0;
118 int h265_pps = 0;
119
120 if (!has_annexb_start_code(data, size, &nal_offset) ||
121 nal_offset >= size)
122 return 0;
123
124 for (size_t pos = 0; pos + 4 <= size; pos++) {
125 if (!annexb_start_code_at(data, size, pos, &nal_offset) ||
126 nal_offset >= size)
127 continue;
128
129 uint8_t h264_type = data[nal_offset] & 0x1f;
130 uint8_t h265_type = (data[nal_offset] >> 1) & 0x3f;
131
132 h264_sps |= (h264_type == 7);
133 h264_pps |= (h264_type == 8);
134 h265_vps |= (h265_type == 32);
135 h265_sps |= (h265_type == 33);
136 h265_pps |= (h265_type == 34);
137
138 if ((h264_sps && h264_pps) ||
139 (h265_vps && (h265_sps || h265_pps)) ||
140 (h265_sps && h265_pps))
141 return 1;
142
143 pos = nal_offset;
144 }
145
146 return 0;
147}

149int rkvc_buffer_looks_compressed_video(const uint8_t *data, size_t size)
150{
151 if (!data || size < 4)
152 return 0;
153
154 if (is_h264_or_h265_annexb(data, size))
155 return 1;
156
157 if (size >= 12 && memcmp(data + 4, "ftyp", 4) == 0)
158 return 1;
159
160 if (size >= 4 && data[0] == 0x1a && data[1] == 0x45 &&
161 data[2] == 0xdf && data[3] == 0xa3)
162 return 1;
163
164 if (size > 188 && data[0] == 0x47 && data[188] == 0x47)
165 return 1;
166
167 return 0;
168}

184rkvc_err rkvc_from_averror(int av_err)
185{
186 if (av_err >= 0)
187 return RKVC_OK;
188
189 switch (av_err) {
190 case AVERROR(ENOMEM): return RKVC_ERR_NOMEM;
191 case AVERROR(EINVAL): return RKVC_ERR_INVALID;
192 case AVERROR(ENOENT): return RKVC_ERR_NOT_FOUND;
193 case AVERROR(EIO): return RKVC_ERR_IO;
194 case AVERROR(EACCES): return RKVC_ERR_PERMISSION;
195 case AVERROR(EPERM): return RKVC_ERR_PERMISSION;
196 case AVERROR_EOF: return RKVC_ERR_EOF;
197 case AVERROR(EAGAIN): return RKVC_ERR_AGAIN;
198 case AVERROR(ENODEV): return RKVC_ERR_HW;
199 case AVERROR_UNKNOWN: return RKVC_ERR_INTERNAL;
200 default:
201 /* AVERROR_DECODER_NOT_FOUND / ENOMEM 等 */
202 if (av_err == AVERROR_DECODER_NOT_FOUND ||
203 av_err == AVERROR_ENCODER_NOT_FOUND)
204 return RKVC_ERR_NOT_FOUND;
205 return RKVC_ERR_INTERNAL;
206 }
207}

209/* ── RKMPP 硬件设备上下文单例 ──────────────────────────────────────── */

211static AVBufferRef *s_hw_device_ctx = NULL;
212static pthread_mutex_t s_hw_lock = PTHREAD_MUTEX_INITIALIZER;
213static int s_hw_init_ok = 0;
214
215rkvc_err rkvc_get_hw_device_ctx(AVBufferRef **out)
216{
217 if (!out)
218 return RKVC_ERR_INVALID;
219
220 pthread_mutex_lock(&s_hw_lock);
221
222 if (!s_hw_init_ok) {
223 int ret = av_hwdevice_ctx_create(&s_hw_device_ctx,
224 AV_HWDEVICE_TYPE_RKMPP,
225 NULL, NULL, 0);
226 if (ret < 0) {
227 RKVC_LOG("RKMPP hw device create failed: %s", av_err2str(ret));
228 s_hw_device_ctx = NULL;
229 pthread_mutex_unlock(&s_hw_lock);
230 return rkvc_from_averror(ret);
231 }
232 s_hw_init_ok = 1;
233 RKVC_LOG("RKMPP hw device created");
234 }
235
236 *out = av_buffer_ref(s_hw_device_ctx);
237 pthread_mutex_unlock(&s_hw_lock);
238
239 if (!*out)
240 return RKVC_ERR_NOMEM;
241 return RKVC_OK;
242}
