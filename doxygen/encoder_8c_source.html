41static const char *guess_muxer(const char *path)
42{
43 const char *dot = strrchr(path, '.');
44 if (!dot)
45 return "hevc";
46
47 dot++;
48 if (strcasecmp(dot, "mp4") == 0) return "mp4";
49 if (strcasecmp(dot, "mkv") == 0) return "matroska";
50 if (strcasecmp(dot, "ts") == 0) return "mpegts";
51 if (strcasecmp(dot, "265") == 0) return "hevc";
52 if (strcasecmp(dot, "h265") == 0) return "hevc";
53 if (strcasecmp(dot, "hevc") == 0) return "hevc";
54 return "hevc";
55}

66static rkvc_err validate_encoder_config(const rkvc_encoder_config *cfg)
67{
68 if (!cfg)
69 return RKVC_ERR_INVALID;
70
71 if (cfg->width <= 0 || cfg->height <= 0)
72 return RKVC_ERR_INVALID;
73 if (cfg->fps_num <= 0 || cfg->fps_den <= 0)
74 return RKVC_ERR_INVALID;
75 if (cfg->bitrate <= 0 || cfg->gop_size <= 0)
76 return RKVC_ERR_INVALID;
77 if (!rkvc_is_valid_pix_fmt(cfg->input_format) ||
78 !rkvc_is_valid_preset(cfg->preset) ||
79 !rkvc_is_valid_rc_mode(cfg->rc_mode))
80 return RKVC_ERR_INVALID;
81 if (cfg->qp < 0 || cfg->num_b_frames < 0 || cfg->threads < 0)
82 return RKVC_ERR_INVALID;
83
84 return RKVC_OK;
85}

94static rkvc_err setup_codec(AVCodecContext *ctx,
95 const rkvc_encoder_config *cfg)
96{
97 ctx->width = cfg->width;
98 ctx->height = cfg->height;
99 ctx->time_base = (AVRational){cfg->fps_den, cfg->fps_num};
100 ctx->framerate = (AVRational){cfg->fps_num, cfg->fps_den};
101 ctx->bit_rate = cfg->bitrate;
102 ctx->gop_size = cfg->gop_size;
103
104 /* sw_pix_fmt 用于软件帧输入; pix_fmt 由 hw_frames_ctx 决定 */
105 ctx->sw_pix_fmt = rkvc_to_av_pix_fmt(cfg->input_format);
106 if (ctx->hw_frames_ctx)
107 ctx->pix_fmt = AV_PIX_FMT_RKMPP;
108 else
109 ctx->pix_fmt = ctx->sw_pix_fmt;
110
111 ctx->max_b_frames = cfg->num_b_frames;
112
113 if (cfg->profile > 0)
114 ctx->profile = cfg->profile;
115 if (cfg->level > 0)
116 ctx->level = cfg->level;
117 if (cfg->threads > 0)
118 ctx->thread_count = cfg->threads;
119
120 /* 编码器私有选项 (通过 AVOption 设置, 忽略不支持的选项) */
121 switch (cfg->preset) {
122 case RKVC_PRESET_FAST:
123 set_codec_option(ctx, "preset", "fast");
124 break;
125 case RKVC_PRESET_SLOW:
126 set_codec_option(ctx, "preset", "slow");
127 break;
128 default:
129 set_codec_option(ctx, "preset", "medium");
130 break;
131 }
132
133 switch (cfg->rc_mode) {
134 case RKRC_VBR:
135 set_codec_option(ctx, "rc_mode", "VBR");
136 break;
137 case RKRC_CQP:
138 set_codec_option(ctx, "rc_mode", "CQP");
139 ctx->global_quality = cfg->qp;
140 break;
141 default:
142 set_codec_option(ctx, "rc_mode", "CBR");
143 break;
144 }
145
146 return RKVC_OK;
147}

159static rkvc_err create_hw_frames_ctx(AVCodecContext *ctx,
160 AVBufferRef *hw_device_ctx,
161 int width, int height,
162 enum AVPixelFormat sw_fmt)
163{
164 AVBufferRef *hw_frames_ref = NULL;
165 AVHWFramesContext *hw_frames_ctx;
166 int ret;
167
168 /*
169 * RKMPP 编码器使用 DRM prime 帧。
170 * 创建硬件帧上下文，sw_format 设为源像素格式 (通常 NV12)。
171 * 编码器内部完成 MPP buffer 转换。
172 */
173 hw_frames_ref = av_hwframe_ctx_alloc(hw_device_ctx);
174 if (!hw_frames_ref)
175 return RKVC_ERR_NOMEM;
176
177 hw_frames_ctx = (AVHWFramesContext *)(hw_frames_ref->data);
178 hw_frames_ctx->format = AV_PIX_FMT_RKMPP;
179 hw_frames_ctx->sw_format = sw_fmt;
180 hw_frames_ctx->width = width;
181 hw_frames_ctx->height = height;
182 hw_frames_ctx->initial_pool_size = 32;
183
184 ret = av_hwframe_ctx_init(hw_frames_ref);
185 if (ret < 0) {
186 av_buffer_unref(&hw_frames_ref);
187 RKVC_LOG("hw frame ctx init failed: %s", av_err2str(ret));
188 return rkvc_from_averror(ret);
189 }
190
191 ctx->hw_frames_ctx = hw_frames_ref;
192 ctx->pix_fmt = AV_PIX_FMT_RKMPP;
193
194 return RKVC_OK;
195}

199static rkvc_err buffer_plane_layout(const rkvc_encoder_config *cfg,
200 int linesize,
201 int plane_heights[4],
202 int copy_bytes[4],
203 int src_linesizes[4])
204{
205 enum AVPixelFormat fmt = rkvc_to_av_pix_fmt(cfg->input_format);
206 int width = cfg->width;
207 int height = cfg->height;
208
209 if (linesize <= 0)
210 return RKVC_ERR_INVALID;
211
212 memset(plane_heights, 0, sizeof(int) * 4);
213 memset(copy_bytes, 0, sizeof(int) * 4);
214 memset(src_linesizes, 0, sizeof(int) * 4);
215
216 switch (fmt) {
217 case AV_PIX_FMT_NV12:
218 if (linesize < width)
219 return RKVC_ERR_INVALID;
220 plane_heights[0] = height;
221 plane_heights[1] = height / 2;
222 copy_bytes[0] = width;
223 copy_bytes[1] = width;
224 src_linesizes[0] = linesize;
225 src_linesizes[1] = linesize;
226 break;
227 case AV_PIX_FMT_YUV420P:
228 if (linesize < width)
229 return RKVC_ERR_INVALID;
230 plane_heights[0] = height;
231 plane_heights[1] = height / 2;
232 plane_heights[2] = height / 2;
233 copy_bytes[0] = width;
234 copy_bytes[1] = width / 2;
235 copy_bytes[2] = width / 2;
236 src_linesizes[0] = linesize;
237 src_linesizes[1] = linesize / 2;
238 src_linesizes[2] = linesize / 2;
239 break;
240 case AV_PIX_FMT_NV16:
241 if (linesize < width)
242 return RKVC_ERR_INVALID;
243 plane_heights[0] = height;
244 plane_heights[1] = height;
245 copy_bytes[0] = width;
246 copy_bytes[1] = width;
247 src_linesizes[0] = linesize;
248 src_linesizes[1] = linesize;
249 break;
250 case AV_PIX_FMT_P010:
251 if (linesize < width * 2)
252 return RKVC_ERR_INVALID;
253 plane_heights[0] = height;
254 plane_heights[1] = height / 2;
255 copy_bytes[0] = width * 2;
256 copy_bytes[1] = width * 2;
257 src_linesizes[0] = linesize;
258 src_linesizes[1] = linesize;
259 break;
260 default:
261 return RKVC_ERR_INVALID;
262 }
263
264 return RKVC_OK;
265}

267static rkvc_err encoder_open_internal(rkvc_encoder **out,
268 const rkvc_encoder_config *cfg,
269 const char *output_path)
270{
271 if (!out)
272 return RKVC_ERR_INVALID;
273
274 *out = NULL;
275
276 rkvc_err err = validate_encoder_config(cfg);
277 if (err != RKVC_OK)
278 return err;
279
280 rkvc_init();
281
282 rkvc_encoder *enc = rkvc_calloc(1, sizeof(*enc));
283 if (!enc)
284 return RKVC_ERR_NOMEM;
285
286 enc->config = *cfg;
287 pthread_mutex_init(&enc->lock, NULL);
288
289 /* 获取硬件设备上下文 */
290 err = rkvc_get_hw_device_ctx(&enc->hw_device_ctx);
291 if (err != RKVC_OK) {
292 RKVC_LOG("hw device ctx failed, falling back to software");
293 enc->hw_device_ctx = NULL;
294 }
295
296 /* 查找编码器 */
297 const AVCodec *codec = avcodec_find_encoder_by_name("hevc_rkmpp");
298 if (!codec) {
299 codec = avcodec_find_encoder(AV_CODEC_ID_HEVC);
300 if (!codec) {
301 rkvc_encoder_close(enc);
302 return RKVC_ERR_NOT_FOUND;
303 }
304 RKVC_LOG("hevc_rkmpp not found, using software: %s", codec->name);
305 }
306
307 /* 创建编码器上下文 */
308 enc->codec_ctx = avcodec_alloc_context3(codec);
309 if (!enc->codec_ctx) {
310 rkvc_encoder_close(enc);
311 return RKVC_ERR_NOMEM;
312 }
313
314 /* 创建硬件帧上下文 (如果有硬件设备) */
315 if (enc->hw_device_ctx) {
316 err = create_hw_frames_ctx(enc->codec_ctx, enc->hw_device_ctx,
317 cfg->width, cfg->height,
318 rkvc_to_av_pix_fmt(cfg->input_format));
319 if (err != RKVC_OK) {
320 RKVC_LOG("hw frames ctx failed, software encode only");
321 av_buffer_unref(&enc->hw_device_ctx);
322 enc->hw_device_ctx = NULL;
323 }
324 }
325
326 /* 配置编码器参数 */
327 err = setup_codec(enc->codec_ctx, cfg);
328 if (err != RKVC_OK) {
329 rkvc_encoder_close(enc);
330 return err;
331 }
332
333 /* 打开编码器 */
334 int ret = avcodec_open2(enc->codec_ctx, codec, NULL);
335 if (ret < 0) {
336 RKVC_LOG("avcodec_open2 failed: %s", av_err2str(ret));
337 rkvc_encoder_close(enc);
338 return rkvc_from_averror(ret);
339 }
340
341 /* 分配 packet */
342 enc->pkt = av_packet_alloc();
343 if (!enc->pkt) {
344 rkvc_encoder_close(enc);
345 return RKVC_ERR_NOMEM;
346 }
347
348 /* 文件模式: 创建 muxer */
349 if (output_path) {
350 const char *muxer_name = guess_muxer(output_path);
351 ret = avformat_alloc_output_context2(&enc->fmt_ctx, NULL,
352 muxer_name, output_path);
353 if (ret < 0) {
354 rkvc_encoder_close(enc);
355 return rkvc_from_averror(ret);
356 }
357
358 enc->av_stream = avformat_new_stream(enc->fmt_ctx, NULL);
359 if (!enc->av_stream) {
360 rkvc_encoder_close(enc);
361 return RKVC_ERR_MUX;
362 }
363
364 /* 从编码器复制参数到流 */
365 ret = avcodec_parameters_from_context(enc->av_stream->codecpar,
366 enc->codec_ctx);
367 if (ret < 0) {
368 rkvc_encoder_close(enc);
369 return rkvc_from_averror(ret);
370 }
371 enc->av_stream->time_base = enc->codec_ctx->time_base;
372
373 /* 打开输出文件 */
374 if (!(enc->fmt_ctx->oformat->flags & AVFMT_NOFILE)) {
375 ret = avio_open(&enc->fmt_ctx->pb, output_path, AVIO_FLAG_WRITE);
376 if (ret < 0) {
377 rkvc_encoder_close(enc);
378 return rkvc_from_averror(ret);
379 }
380 }
381
382 /* 写文件头 */
383 ret = avformat_write_header(enc->fmt_ctx, NULL);
384 if (ret < 0) {
385 rkvc_encoder_close(enc);
386 return rkvc_from_averror(ret);
387 }
388
389 enc->file_mode = 1;
390 }
391
392 *out = enc;
393 return RKVC_OK;
394}

413rkvc_err rkvc_encoder_send_frame(rkvc_encoder *enc, rkvc_frame *f)
414{
415 if (!enc || !enc->codec_ctx)
416 return RKVC_ERR_INVALID;
417
418 if (enc->flushed)
419 return RKVC_ERR_EOF;
420
421 AVFrame *frame = NULL;
422 if (f) {
423 frame = f->av_frame;
424 /*
425 * RKMPP 编码器要求 DMA 硬件帧 (linear / AFBC)。
426 * 将软件 NV12 帧上传到硬件表面再送入编码器。
427 */
428 if (enc->codec_ctx->hw_frames_ctx &&
429 frame->format != AV_PIX_FMT_RKMPP &&
430 frame->format != AV_PIX_FMT_DRM_PRIME) {
431 AVFrame *hw_frame = av_frame_alloc();
432 if (!hw_frame)
433 return RKVC_ERR_NOMEM;
434
435 hw_frame->format = AV_PIX_FMT_RKMPP;
436 int ret = av_hwframe_get_buffer(enc->codec_ctx->hw_frames_ctx,
437 hw_frame, 0);
438 if (ret < 0) {
439 av_frame_free(&hw_frame);
440 return rkvc_from_averror(ret);
441 }
442
443 ret = av_hwframe_transfer_data(hw_frame, frame, 0);
444 if (ret < 0) {
445 av_frame_free(&hw_frame);
446 return rkvc_from_averror(ret);
447 }
448
449 hw_frame->pts = frame->pts;
450 frame = hw_frame;
451 }
452 }
453
454 int ret = avcodec_send_frame(enc->codec_ctx, frame);
455
456 /* 编码器内部会引用 hw_frame，此处释放我们的引用 */
457 if (f && frame != f->av_frame)
458 av_frame_free(&frame);
459
460 if (ret < 0) {
461 if (ret == AVERROR(EAGAIN))
462 return RKVC_ERR_AGAIN;
463 return rkvc_from_averror(ret);
464 }
465
466 return RKVC_OK;
467}

469rkvc_err rkvc_encoder_send_buffer(rkvc_encoder *enc,
470 const uint8_t *data, int linesize,
471 int64_t pts)
472{
473 if (!enc || !data)
474 return RKVC_ERR_INVALID;
475
476 int plane_heights[4];
477 int copy_bytes[4];
478 int src_linesize[4];
479 rkvc_err err = buffer_plane_layout(&enc->config, linesize,
480 plane_heights, copy_bytes,
481 src_linesize);
482 if (err != RKVC_OK)
483 return err;
484
485 rkvc_frame *f = NULL;
486 err = rkvc_frame_alloc(&f, enc->config.width,
487 enc->config.height,
488 enc->config.input_format);
489 if (err != RKVC_OK)
490 return err;
491
492 /* 复制像素数据 */
493 AVFrame *dst = f->av_frame;
494 const uint8_t *src_data[4] = {NULL, NULL, NULL, NULL};
495
496 src_data[0] = data;
497 for (int i = 1; i < 4 && plane_heights[i] > 0; i++) {
498 src_data[i] = src_data[i - 1] +
499 src_linesize[i - 1] * plane_heights[i - 1];
500 }
501
502 for (int i = 0; i < 4 && plane_heights[i] > 0; i++) {
503 if (dst->data[i]) {
504 for (int row = 0; row < plane_heights[i]; row++) {
505 memcpy(dst->data[i] + row * dst->linesize[i],
506 src_data[i] + row * src_linesize[i],
507 copy_bytes[i]);
508 }
509 }
510 }
511
512 dst->pts = pts;
513 f->info.pts = pts;
514
515 err = rkvc_encoder_send_frame(enc, f);
516 rkvc_frame_unref(f);
517 return err;
518}

522rkvc_err rkvc_encoder_receive_packet(rkvc_encoder *enc, rkvc_packet *pkt)
523{
524 if (!enc || !enc->codec_ctx || !pkt)
525 return RKVC_ERR_INVALID;
526
527 memset(pkt, 0, sizeof(*pkt));
528
529 int ret = avcodec_receive_packet(enc->codec_ctx, enc->pkt);
530 if (ret < 0) {
531 if (ret == AVERROR(EAGAIN))
532 return RKVC_ERR_AGAIN;
533 if (ret == AVERROR_EOF)
534 return RKVC_ERR_EOF;
535 return rkvc_from_averror(ret);
536 }
537
538 /* 文件模式: 写入封装 */
539 if (enc->file_mode && enc->fmt_ctx) {
540 av_packet_rescale_ts(enc->pkt, enc->codec_ctx->time_base,
541 enc->av_stream->time_base);
542 enc->pkt->stream_index = enc->av_stream->index;
543
544 ret = av_interleaved_write_frame(enc->fmt_ctx, enc->pkt);
545 if (ret < 0) {
546 av_packet_unref(enc->pkt);
547 return RKVC_ERR_MUX;
548 }
549 }
550
551 /* 提取元数据到输出结构 */
552 pkt->data = enc->pkt->data;
553 pkt->size = enc->pkt->size;
554 pkt->pts = enc->pkt->pts;
555 pkt->dts = enc->pkt->dts;
556 pkt->key_frame = (enc->pkt->flags & AV_PKT_FLAG_KEY) ? 1 : 0;
557 pkt->pos = enc->pkt->pos;
558
559 if (enc->file_mode) {
560 /* 文件模式: av_interleaved_write_frame 已消费数据，清理 */
561 av_packet_unref(enc->pkt);
562 }
563 /* 非文件模式: enc->pkt 保持有效，数据在下次 send_frame/receive 前可用 */
564
565 return RKVC_OK;
566}

587/* ── 关闭 ──────────────────────────────────────────────────────────── */
588
589rkvc_err rkvc_encoder_close(rkvc_encoder *enc)
590{
591 if (!enc)
592 return RKVC_OK;
593
594 /* 自动 drain (如果尚未 drain 且有待处理的帧) */
595 if (enc->codec_ctx && !enc->flushed) {
596 rkvc_encoder_drain(enc);
597
598 /* flush 残余包 */
599 while (avcodec_receive_packet(enc->codec_ctx, enc->pkt) >= 0) {
600 if (enc->file_mode && enc->fmt_ctx) {
601 av_packet_rescale_ts(enc->pkt, enc->codec_ctx->time_base,
602 enc->av_stream->time_base);
603 enc->pkt->stream_index = enc->av_stream->index;
604 av_interleaved_write_frame(enc->fmt_ctx, enc->pkt);
605 }
606 av_packet_unref(enc->pkt);
607 }
608 }
609
610 /* 文件模式: 写 trailer */
611 if (enc->file_mode && enc->fmt_ctx) {
612 av_write_trailer(enc->fmt_ctx);
613 }
614
615 /* 释放资源 */
616 if (enc->pkt)
617 av_packet_free(&enc->pkt);
618 if (enc->codec_ctx) {
619 avcodec_free_context(&enc->codec_ctx);
620 }
621 if (enc->fmt_ctx) {
622 if (!(enc->fmt_ctx->oformat->flags & AVFMT_NOFILE) && enc->fmt_ctx->pb)
623 avio_closep(&enc->fmt_ctx->pb);
624 avformat_free_context(enc->fmt_ctx);
625 }
626 if (enc->hw_device_ctx)
627 av_buffer_unref(&enc->hw_device_ctx);
628
629 pthread_mutex_destroy(&enc->lock);
630 rkvc_free(enc);
631 return RKVC_OK;
632}
